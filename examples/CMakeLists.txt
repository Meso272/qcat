function(build_example)
  set(source_file ${ARGV0})
  get_filename_component(bin_name "${source_file}" NAME_WLE)
  add_executable(${bin_name} ${source_file})
  target_link_libraries(${bin_name} qcat)
  install(TARGETS ${bin_name})
endfunction()
